void move(int num){
	int i;
	for(i=0 ; i<8 ; ++i){
		if(i>0)
		display[i-1] = display[i] ;
	}
	if( num > 25 ) num = 0;
	display[7] = num1[num];
}

void show(){
	int j;
	for(j=0 ; j<8 ; ++j){
		P1 = 0xff;
		P2 = display[j];
		P1 = line2[j];
		delay(1);
	}
}
void delay(int n){
	int i ,j;
	for( i=0 ; i<n ; ++i)
		for( j=0 ; j<50 ; ++j);
}
