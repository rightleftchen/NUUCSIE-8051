#include <reg51.h>
#include <string.h>
#define DBPORT P2
sbit RS=P1^0;
sbit RW=P1^1;
sbit E=P1^2;
void delay1ms(int count);
void Init_LCD();
void LCD_CMD(char cmd);
void LCD_DATA(char data1);
void DISP_Str(char addr1,char *str);
void DISP_Int(char addr1,int v1);
void RL_Str(char addr1,char *str,char offset);


char code *mes[]={"1111111",
								"2222222",
								"3333333"};
main()
{
	char i;
	Init_LCD();
	LCD_CMD(1);
	delay1ms(10);
	//DISP_Str(0x80,mes1);	
	while(1){
		for(i=0;i<3;i++)
		{
			DISP_Str(0x80,mes[i]);
			DISP_Str(0xC0,mes[(i+1)%3]);
			delay1ms(1000);
		}
		
  }		
	
}
void Init_LCD()
{
	delay1ms(5);
	LCD_CMD(0x3f);
	LCD_CMD(0xe);
	LCD_CMD(0x1);
	delay1ms(2);
	LCD_CMD(0x80);
}
void LCD_CMD(char cmd)
{
	char i;
	DBPORT=cmd;
	RS=0;RW=0;E=1;
	for(i=0;i<40;i++)
	RS=0;RW=0;E=0;
}
void LCD_DATA(char data1)
{
	char i;
	DBPORT=data1;
	RS=1;RW=0;E=1;
	for(i=0;i<40;i++)
	RS=1;RW=0;E=0;
}
void delay1ms(int count)
{
	int i,j;
	for(i=0;i<count;i++)
	for(j=0;j<150;j++);
}
void DISP_Str(char addr1,char *str)
{
	LCD_CMD(addr1);
	while(*str!=0)
		LCD_DATA(*str++);
}
void DISP_Int(char addr1,int v1)
{
	char i;
	for(i=0;i<4;i++)
	{
		LCD_CMD(addr1-i+4);
		LCD_DATA((v1%10)+0x30);
		v1/=10;
		if(v1==0)
			break;
	}	
}
void RL_Str(char addr1,char *str,char offset) // display 
{
	char i;
	char *start=str;
	str+=offset;
	LCD_CMD(addr1);
	while(*str!=0)
		LCD_DATA(*str++);
	for(i=0;i<offset;i++)
		LCD_DATA(start[i]);
}
void RR_Str(char addr1,char *str,char offset) // display 
{
	char i;
	char *start=str;
	LCD_CMD(addr1);
	str+=20-offset;	
	while(*str!=0)
		LCD_DATA(*str++);
	for(i=0;i<offset;i++)
		LCD_DATA(start[i]);
}
